13  """
14  A class providing a high-level interface to
15  a client connected via the NEATClient.
16  It receives commands from NEATServer's message
17  queue und responds to them.
18  When a function from SimulationConnector is called,
19  the client is expected to have sent a certain command
20  (if there are no commands in the message queue, the Server
21  will wait for the next command to be received). If the
22  client has sent another command, or a command where
23  the expected parameters are not present,
24  NetworkProtocolException will be raised.
25  """

30  def _listen_for_command(
31  self,
32  command_type: str,
33  parameter_filter: Dict[str, object]=None,
34  timeout: int = None
35  ) -> BaseCommand:
36  """
37  Waits for the message queue to have a command ready,
38  pops the command from the queue and inspects it.
39  Returns the received command if it's type and parameters
40  match type / filter.
41  If the next received command does not match,
42  NetworkProtocolException is raised.
43  :param command_type: The type of the expected command as string
44  :param parameter_filter: A dictionary of parameters that are expected to be
45  present in the command's parameters.
46  :param timeout: The timeout after which the attempt to get
47  the command will fail
48  :return: The received command object
49  """
50  # TODO: timeouts, more error handling
51  command = self._server.fetch(timeout)
52  print("Received: ", command)
53  if not command:
54  raise NetworkTimeoutException
55 
56  if not command.type == command_type:
57  self._respond_to_command(command, acknowledged=False)
58  raise NetworkProtocolException("Wrong command type encountered")
59 
60  if parameter_filter:
61  for key, value in parameter_filter.items():
62  if key not in command.parameters.keys():
63  self._respond_to_command(command, acknowledged=False)
64  raise NetworkProtocolException("Filter key not in command parameters")
65  if not value == command.parameters[key]:
66  self._respond_to_command(command, acknowledged=False)
67  raise NetworkProtocolException("Filter values do not match in command parameters")
68 
69  return command

71  def _respond_to_command(
72  self,
73  command: BaseCommand,
74  result: Dict[str, object]=None,
75  acknowledged: bool=True,
76  timeout: int=2000
77  ) -> None:
78  """
79  Sends the altered command object of a previously
80  received command back to the client.
81  The command object that is sent back to the client
82  should always be the same object that had been received
83  from the client.
84  the results of the command execution are passed back to client
85  inside the command.result dictionary. If this dictionary
86  isn't present in the passed command, it will be created.
87  All commands that're passed back to the client and that
88  have been handled by the server contain an 'acknowledged'
89  field inside results. If this is set to true, the command
90  has been executed and the networking protocol hasn't been
91  breached.
92  :param command: The command object that is sent back
93  to the client
94  :param result: The result dictionary to be attached to
95  the command before it's sent back
96  :param acknowledged: Whether the command has been
97  acknowledged by the server
98  :param timeout: The timeout after which the send operation
99  will fail if the client hasn't connected yet
100  :return: None
101  """
102  if result is None:
103  result = dict({})
104  command.result = result
105  command.result["acknowledged"] = acknowledged
106  self._server.respond(command)
